// ── Date + day counter ──
function updateDate() {
  const now = new Date();
  const dayNum = Math.max(1, Math.floor((now - START_DATE) / 86400000) + 1);
  const weekNum = Math.min(8, Math.ceil(dayNum / 7));
  document.getElementById('dateBadge').innerHTML =
    now.toLocaleDateString('en-US', {month:'short',day:'numeric'}) +
    `<br/><span>Week ${weekNum} · Day ${dayNum}</span>`;
}


// ── Today verse + motive ──
function loadToday() {
  const idx = new Date().getDay(); // 0–6
  const v = DAILY_VERSES[idx % DAILY_VERSES.length];
  const m = DAILY_MOTIVES[idx % DAILY_MOTIVES.length];
  document.getElementById('todayVerse').textContent = `"${v.text}"`;
  document.getElementById('todayVerseRef').textContent = v.ref;
  document.getElementById('todayMotive').textContent = `"${m.text}"`;
  document.getElementById('todayMotiveAttr').textContent = `— ${m.attr}`;


  // Today's workout
  const now = new Date();
  const dayNum = Math.max(0, Math.floor((now - START_DATE) / 86400000));
  const weekIdx = Math.min(7, Math.floor(dayNum / 7));
  const dayOfWeek = now.getDay(); // 0=Sun
  const dayMap = [6,0,1,2,3,4,5]; // Sun→index6, Mon→0...
  const dIdx = dayMap[dayOfWeek];
  const todayDay = WEEKS[weekIdx].days[dIdx];
  const wkLabel = document.getElementById('todayWorkout');
  if (todayDay) {
    const typeColor = todayDay.type === 'strength' ? 'var(--amber)' : todayDay.type === 'cardio' ? 'var(--teal)' : 'var(--muted)';
    let html = `<div style="color:${typeColor};font-weight:700;font-size:14px;margin-bottom:4px;">${todayDay.label}</div>`;
    if (todayDay.note) html += `<div style="font-size:11px;color:var(--muted);margin-bottom:8px;">${todayDay.note}</div>`;
    if (todayDay.cardio) {
      html += `<div style="font-size:12px;color:var(--cream);">📏 ${todayDay.cardio.dist}<br/>⏱ ${todayDay.cardio.pace}<br/><span style="color:var(--muted);">📍 ${todayDay.cardio.loc}</span></div>`;
    }
    if (todayDay.exercises.length) {
      todayDay.exercises.slice(0,2).forEach(ex => {
        html += `<div style="margin-top:8px;font-size:10px;color:var(--amber);letter-spacing:2px;">${ex.group.toUpperCase()}</div>`;
        html += `<div style="font-size:12px;color:var(--cream);">${ex.opts[0].replace('★ ','')}</div>`;
      });
    }
    wkLabel.innerHTML = html;
  }
}


// ── Plan page ──
let currentWeek = 0;
function buildPlan() {
  const scroller = document.getElementById('weekScroller');
  scroller.innerHTML = WEEKS.map((w,i) =>
    `<button class="week-pill${i===0?' active':''}" onclick="selectWeek(${i})" id="pill-${i}">${i+1}</button>`
  ).join('');
  renderWeek(0);
}


function selectWeek(i) {
  currentWeek = i;
  document.querySelectorAll('.week-pill').forEach(p => p.classList.remove('active'));
  document.getElementById('pill-' + i).classList.add('active');
  renderWeek(i);
}


function renderWeek(i) {
  const w = WEEKS[i];
  let html = `<div class="week-theme">
    <div class="week-name">${w.label}</div>
    <div class="week-desc">${w.theme}</div>
    <div class="hume-box"><div class="hume-label">⌚ HUME BAND</div><div class="hume-text">${w.hume}</div></div>
  </div>`;


  w.days.forEach((day, di) => {
    const tagClass = day.type === 'strength' ? 'tag-strength' : day.type === 'cardio' ? 'tag-cardio' : 'tag-rest';
    const typeEmoji = day.type === 'strength' ? '💪' : day.type === 'cardio' ? '🏃' : '😴';
    html += `<div class="day-row">
      <button class="day-header" onclick="toggleDay('d-${i}-${di}')">
        <span class="day-tag ${tagClass}">${day.day}</span>
        <span>
          <div class="day-title">${day.label}</div>
          <div class="day-sub">${typeEmoji} ${day.type.toUpperCase()}</div>
        </span>
        <span class="day-chevron" id="chev-d-${i}-${di}">›</span>
      </button>
      <div class="day-body" id="d-${i}-${di}">`;
    if (day.note) html += `<div class="day-note">${day.note}</div>`;
    if (day.cardio) {
      html += `<div class="cardio-box">
        <div class="cardio-title">${day.cardio.type}</div>
        <div class="cardio-row">📏 ${day.cardio.dist}</div>
        <div class="cardio-row">⏱ ${day.cardio.pace}</div>
        <div class="cardio-loc">📍 ${day.cardio.loc}</div>
      </div>`;
    }
    day.exercises.forEach(ex => {
      html += `<div class="exercise-group"><div class="ex-group-label">${ex.group}</div>`;
      ex.opts.forEach((opt,oi) => {
        const isPrimary = opt.startsWith('★');
        const label = opt.replace('★ ','').replace('○ ','');
        html += `<div class="ex-option">
          <span class="ex-star">${isPrimary ? '★' : '○'}</span>
          <span class="ex-text ${isPrimary ? 'primary' : 'alt'}">${isPrimary ? '<strong>'+label+'</strong>' : label}</span>
        </div>`;
      });
      html += `</div>`;
    });
    html += `</div></div>`;
  });
  document.getElementById('weekContent').innerHTML = html;
}


function toggleDay(id) {
  const body = document.getElementById(id);
  const chev = document.getElementById('chev-' + id);
  const isOpen = body.classList.contains('open');
  body.classList.toggle('open', !isOpen);
  chev.classList.toggle('open', !isOpen);
}


// ── Copy text ──
function copyText(el) {
  const text = el.querySelector('div:last-child').textContent;
  navigator.clipboard.writeText(text).then(() => {
    const label = el.querySelector('div:first-child');
    const orig = label.textContent;
    label.textContent = '✓ COPIED!';
    setTimeout(() => label.textContent = orig, 1500);
  });
}
